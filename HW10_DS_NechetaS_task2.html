def loadMovieList():
    """
    Reads the fixed movie list in movie_ids.txt and returns a list of movie names.
    Returns
    -------
    movieNames : list
        A list of strings, representing all movie names.
    """
    # Read the fixed movie list
    file_path = r"D:\PYTHON_LEARNING\HW_10_ML\movie_ids.txt"
    with open(file_path, "r", encoding='ISO-8859-1') as fid:
        movies = fid.readlines()

    movieNames = []
    for movie in movies:
        parts = movie.split()
        movieNames.append(' '.join(parts[1:]).strip())

    return movieNames

def cofiCostFuncReg(params, Y, R, num_users, num_movies, num_features, lmbda):
    # Розпаковка X та W
    X = params[:num_movies*num_features].reshape((num_movies, num_features))
    W = params[num_movies*num_features:].reshape((num_users, num_features))
    
    # Векторизоване обчислення помилки тільки там, де R=1
    error = (X @ W.T - Y) * R
    J = 0.5 * np.sum(error**2)
    
    # Додаємо регуляризацію до вартості
    J += (lmbda / 2) * (np.sum(W**2) + np.sum(X**2))
    
    # Регуляризовані градієнти
    X_grad = error @ W + lmbda * X
    W_grad = error.T @ X + lmbda * W
    
    # Згортаємо градієнти в один вектор
    grad = np.concatenate([X_grad.ravel(), W_grad.ravel()])
    
    return J, grad

def costFunc(params):
    J, grad = cofiCostFuncReg(params, Y, R, num_users, num_movies, num_features, lmbda)
    return J, grad
